### 方法签名

<Tabs>
  <Tab title="TypeScript">
    ```typescript
    // With schema and options
    await page.extract<T extends z.AnyZodObject>(options: ExtractOptions<T>): Promise<ExtractResult<T>>

    // String instruction only
    await page.extract(instruction: string): Promise<{ extraction: string }>

    // No parameters (raw page content)
    await page.extract(): Promise<{ page_text: string }>
    ```

    **ExtractOptions 接口：**

    ```typescript
    interface ExtractOptions<T extends z.AnyZodObject> {
      instruction?: string;
      schema?: T;
      modelName?: AvailableModel;
      modelClientOptions?: ClientOptions;
      domSettleTimeoutMs?: number;
      selector?: string;
      iframes?: boolean;
    }

    type ExtractResult<T> = z.infer<T>;
    ```
  </Tab>

  <Tab title="Python">
    ```python
    # With schema and parameters
    await page.extract(
        instruction: str = None,
        schema: BaseModel = None,
        selector: str = None,
        iframes: bool = None,
        model_name: AvailableModel = None,
        model_client_options: Dict = None,
        dom_settle_timeout_ms: int = None
    ) -> ExtractResult

    # String instruction only
    await page.extract(instruction: str) -> Dict[str, str]

    # No parameters (raw page content)
    await page.extract() -> Dict[str, str]
    ```
  </Tab>
</Tabs>

### 参数

<ParamField path="instruction" type="string" optional>
  用自然语言描述需要提取的数据。
</ParamField>

<ParamField path="schema" type="z.ZodSchema | BaseModel" optional>
  定义要提取数据结构的类型模式。确保类型安全与校验。
</ParamField>

<ParamField path="selector" type="string" optional>
  将提取范围限定在指定区域的 XPath 选择器。可减少 token 使用并提升准确性。
</ParamField>

<ParamField path="iframes" type="boolean" optional>
  若内容位于 iframes 中，请设为 `true`。

  **默认值：** `false`
</ParamField>

<ParamField path="modelName" type="AvailableModel" optional>
  覆盖此次提取使用的默认 LLM 模型。
</ParamField>

<ParamField path="modelClientOptions" type="ClientOptions" optional>
  模型特定的配置项。
</ParamField>

<ParamField path="domSettleTimeoutMs" type="number" optional>
  等待 DOM 稳定的最长时间。

  **默认值：** `30000`
</ParamField>

### 响应类型

<Tabs>
  <Tab title="With Schema">
    **返回：** `Promise<ExtractResult<T>>`，其中 T 与你的 schema 匹配

    返回对象将严格遵循你的 schema 定义进行类型约束。
  </Tab>

  <Tab title="String Only">
    **返回：** `Promise<{ extraction: string }>`
    不经 schema 校验的简单字符串提取。
  </Tab>

  <Tab title="No Parameters">
    **返回：** `Promise<{ page_text: string }>`
    页面内容的原始可访问性树表示。
  </Tab>
</Tabs>

### 代码示例

<Tabs>
  <Tab title="Single Object">
    <CodeGroup>
      ```typescript TypeScript
      import { z } from 'zod';

      // 模式定义
      const ProductSchema = z.object({
        name: z.string(),
        price: z.number(),
        inStock: z.boolean()
      });

      // 数据提取
      const product = await page.extract({
        instruction: "提取产品详情",
        schema: ProductSchema
      });
      ```

      ```python Python
      from pydantic import BaseModel

      # 模式定义
      class Product(BaseModel):
          name: str
          price: float
          in_stock: bool

      # 数据提取
      product = await page.extract(
          instruction="提取产品详情",
          schema=Product
      )
      ```
    </CodeGroup>

    #### 示例响应

    ```json
    {
      "name": "Product Name",
      "price": 100,
    ```

  <Tab title="Advanced">
    <CodeGroup>
      ```typescript TypeScript
      import { z } from 'zod';

      // 带有描述和校验的架构
      const ProductSchema = z.object({
        price: z.number().describe("以 USD 计的商品价格"),
        rating: z.number().min(0).max(5).describe("客户评分（满分 5 分）"),
        available: z.boolean().describe("商品是否有库存"),
        tags: z.array(z.string()).optional()
      });

      // 嵌套架构
      const EcommerceSchema = z.object({
        product: z.object({
          name: z.string(),
          price: z.object({
            current: z.number(),
            original: z.number().optional()
          })
        }),
        reviews: z.array(z.object({
          rating: z.number(),
          comment: z.string()
        }))
      });
      ```

      ```python Python
      from pydantic import BaseModel, Field
      from typing import Optional, List

      # 带有描述和校验的模型
      class Product(BaseModel):
          price: float = Field(description="以 USD 计的商品价格")
          rating: float = Field(ge=0, le=5, description="客户评分（满分 5 分）")
          available: bool = Field(description="商品是否有库存")
          tags: Optional[List[str]] = None

      # 嵌套模型
      class Price(BaseModel):
          current: float
          original: Optional[float] = None

      class Review(BaseModel):
          rating: int
          comment: str

      class ProductDetails(BaseModel):
          name: str
          price: Price

      class EcommerceData(BaseModel):
          product: ProductDetails
          reviews: List[Review]
      ```
    </CodeGroup>
